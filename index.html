searchable-select.js 
class SearchableSelect {
  constructor(el) {
    this.el = el;
    this.trigger = el.querySelector('.select-trigger');
    this.triggerText = el.querySelector('.trigger-text');
    this.searchInput = el.querySelector('.search-input');
    this.options = el.querySelectorAll('.option');
    this.searchCount = el.querySelector('.search-count');
    this.hiddenInput = el.parentElement.querySelector('.select-hidden-input');
    this.isOpen = false;
    this.totalCount = this.options.length;
    this._updateCount(this.totalCount);
    this._bindEvents();
  }

  _updateCount(n) {
    if (this.searchCount) this.searchCount.textContent = `${n}/${this.totalCount}`;
  }

  _bindEvents() {
    this.trigger.addEventListener('click', () => this.toggle());
    this.trigger.addEventListener('keydown', e => {
      if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); this.toggle(); }
    });
    this.searchInput.addEventListener('input', e => this._filter(e.target.value));
    this.options.forEach(opt => opt.addEventListener('click', () => this._selectOption(opt)));
    document.addEventListener('click', e => { if (!this.el.contains(e.target)) this.close(); });
    document.addEventListener('keydown', e => { if (e.key === 'Escape') this.close(); });
  }

  toggle() { this.isOpen ? this.close() : this.open(); }

  open() {
    document.querySelectorAll('.searchable-select.open').forEach(s => {
      if (s !== this.el) s._ssInstance?.close();
    });
    this.el.classList.add('open');
    this.trigger.setAttribute('aria-expanded', 'true');
    this.isOpen = true;
    requestAnimationFrame(() => this.searchInput.focus());
  }

  close() {
    this.el.classList.remove('open');
    this.trigger.setAttribute('aria-expanded', 'false');
    this.searchInput.value = '';
    this._filter('');
    this.isOpen = false;
  }

  _filter(term) {
    const t = term.toLowerCase().trim();
    let visible = 0;
    this.options.forEach(opt => {
      const match = opt.textContent.toLowerCase().includes(t);
      opt.classList.toggle('hidden', !match);
      if (match) visible++;
    });
    this._updateCount(visible);
    this._handleNoResults(visible);
  }

  _handleNoResults(n) {
    let msg = this.el.querySelector('.no-results');
    if (n === 0) {
      if (!msg) {
        msg = document.createElement('li');
        msg.className = 'no-results';
        msg.innerHTML = `
          <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5">
            <circle cx="11" cy="11" r="8"/><line x1="21" y1="21" x2="16.65" y2="16.65"/>
            <line x1="8" y1="8" x2="14" y2="14"/>
          </svg>No results found`;
        this.el.querySelector('.options-list').appendChild(msg);
      }
    } else if (msg) msg.remove();
  }

  _selectOption(opt) {
    this.options.forEach(o => o.classList.remove('selected'));
    opt.classList.add('selected');
    const label = opt.childNodes[0].textContent.trim();
    this.triggerText.textContent = label;
    this.triggerText.classList.remove('placeholder');
    this.el.classList.add('has-value');
    if (this.hiddenInput) this.hiddenInput.value = opt.dataset.value;
    this.close();
  }

  // Public API
  setValue(val) {
    const opt = [...this.options].find(o => o.dataset.value === val);
    if (opt) this._selectOption(opt);
  }

  getValue() {
    return this.hiddenInput ? this.hiddenInput.value : null;
  }

  clear() {
    this.options.forEach(o => o.classList.remove('selected'));
    this.triggerText.textContent = this.trigger.dataset.placeholder || 'Choose…';
    this.triggerText.classList.add('placeholder');
    this.el.classList.remove('has-value');
    if (this.hiddenInput) this.hiddenInput.value = '';
  }
}

// Auto-init
document.addEventListener('DOMContentLoaded', () => {
  document.querySelectorAll('.searchable-select').forEach(el => {
    el._ssInstance = new SearchableSelect(el);
  });
});
Quick Setup
1 Save the CSS and JS, then link them — <link> in <head> and <script> before </body>.
2 Paste the HTML inside any <form>. Add <li class="option" data-value="..."> items inside .options-list.
3 JS auto-initializes every .searchable-select on load. Access programmatically via el._ssInstance.

/**
 * MultiSelect — reads options from a native <select multiple>,
 * builds a fully custom tag-based dropdown.
 *
 * Usage:
 *   <select id="mySelect" multiple>
 *     <option value="1">Option 1</option>
 *   </select>
 *   const ms = new MultiSelect('mySelect', { placeholder: 'Pick...' });
 */
class MultiSelect {
  constructor(selectId, config = {}) {
    this.select = document.getElementById(selectId);
    if (!this.select) throw new Error(`#${selectId} not found`);

    this.config = {
      placeholder:       'Select options...',
      searchPlaceholder: 'Search...',
      showSelectAll:     true,
      maxTags:           3,
      onChange:          null,
      hiddenInputId:     null,
      ...config
    };

    this.options = Array.from(this.select.options).map(o => ({
      value: o.value, label: o.textContent.trim(), selected: o.selected
    }));
    this.filtered    = [...this.options];
    this.isOpen      = false;
    this.highlighted = -1;
    this.hiddenInput = this.config.hiddenInputId
      ? document.getElementById(this.config.hiddenInputId) : null;

    this._build();
    this._bindEvents();
    this._updateUI();
  }

  /* ── Build DOM ── */
  _build() {
    this.wrapper = document.createElement('div');
    this.wrapper.className = 'ms-dropdown';

    // Trigger
    this.trigger = document.createElement('div');
    this.trigger.className = 'ms-trigger';
    this.trigger.setAttribute('tabindex', '0');

    this.selectedBox = document.createElement('div');
    this.selectedBox.className = 'ms-selected';

    this.placeholder = document.createElement('span');
    this.placeholder.className = 'ms-placeholder';
    this.placeholder.textContent = this.config.placeholder;

    this.arrow = document.createElement('div');
    this.arrow.className = 'ms-arrow';
    this.arrow.innerHTML = `<svg width="12" height="12" viewBox="0 0 12 12"
      fill="none" stroke="currentColor" stroke-width="2">
      <path d="M2 4L6 8L10 4"/></svg>`;

    this.trigger.append(this.selectedBox, this.arrow);

    // Panel
    this.panel = document.createElement('div');
    this.panel.className = 'ms-panel';

    // Search
    const sw = document.createElement('div');
    sw.className = 'ms-search';
    sw.innerHTML = `<svg class="ms-search-icon" viewBox="0 0 24 24" fill="none"
      stroke="currentColor" stroke-width="2">
      <circle cx="11" cy="11" r="8"/>
      <line x1="21" y1="21" x2="16.65" y2="16.65"/></svg>`;
    this.searchInput = document.createElement('input');
    this.searchInput.type = 'text';
    this.searchInput.placeholder = this.config.searchPlaceholder;
    this.searchInput.setAttribute('autocomplete', 'off');
    sw.appendChild(this.searchInput);
    this.panel.appendChild(sw);

    // Select All
    if (this.config.showSelectAll) {
      this.selectAllRow = document.createElement('div');
      this.selectAllRow.className = 'ms-select-all';
      this.selectAllBox = document.createElement('input');
      this.selectAllBox.type = 'checkbox';
      this.selectAllRow.append(this.selectAllBox, document.createTextNode(' Select All'));
      this.panel.appendChild(this.selectAllRow);
    }

    // List
    this.list = document.createElement('ul');
    this.list.className = 'ms-list';
    this.panel.appendChild(this.list);

    // Footer
    this.footer = document.createElement('div');
    this.footer.className = 'ms-footer';
    this.panel.appendChild(this.footer);

    this.wrapper.append(this.trigger, this.panel);
    this.select.parentNode.insertBefore(this.wrapper, this.select.nextSibling);
    this.select.style.display = 'none';
  }

  /* ── Events ── */
  _bindEvents() {
    this.trigger.addEventListener('click', e => { e.stopPropagation(); this._toggle(); });
    this.trigger.addEventListener('keydown', e => {
      if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); this._toggle(); }
      if (e.key === 'ArrowDown') { e.preventDefault(); if (!this.isOpen) this._open(); }
      if (e.key === 'Escape') this._close();
    });
    this.searchInput.addEventListener('input', e => this._filter(e.target.value));
    this.searchInput.addEventListener('click', e => e.stopPropagation());
    this.searchInput.addEventListener('keydown', e => {
      e.stopPropagation();
      if (e.key === 'ArrowDown') { e.preventDefault(); this._moveHighlight(1); }
      if (e.key === 'ArrowUp')   { e.preventDefault(); this._moveHighlight(-1); }
      if (e.key === 'Enter')     { e.preventDefault(); this._selectHighlighted(); }
      if (e.key === 'Escape')    { this._close(); this.trigger.focus(); }
    });
    if (this.selectAllRow) {
      this.selectAllRow.addEventListener('click', e => { e.stopPropagation(); this._toggleAll(); });
    }
    document.addEventListener('click', e => { if (!this.wrapper.contains(e.target)) this._close(); });
    this.panel.addEventListener('click', e => e.stopPropagation());
  }

  _toggle() { this.isOpen ? this._close() : this._open(); }

  _open() {
    this.isOpen = true;
    this.trigger.classList.add('open');
    this.panel.classList.add('open');
    this._renderList();
    this._updateSelectAll();
    setTimeout(() => {
      this.searchInput.focus();
      if (this.filtered.length > 0) { this.highlighted = 0; this._updateHighlight(); }
    }, 0);
  }

  _close() {
    this.isOpen = false;
    this.trigger.classList.remove('open');
    this.panel.classList.remove('open');
    this.searchInput.value = '';
    this._filter('');
    this.highlighted = -1;
  }

  _filter(q) {
    const t = q.toLowerCase().trim();
    this.filtered = t ? this.options.filter(o => o.label.toLowerCase().includes(t)) : [...this.options];
    this.highlighted = this.filtered.length > 0 ? 0 : -1;
    this._renderList();
    this._updateSelectAll();
  }

  _renderList() {
    this.list.innerHTML = '';
    if (this.filtered.length === 0) {
      const li = document.createElement('li');
      li.className = 'ms-empty';
      li.innerHTML = `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5">
        <circle cx="11" cy="11" r="8"/><line x1="21" y1="21" x2="16.65" y2="16.65"/>
        <line x1="8" y1="8" x2="14" y2="14"/></svg>No results found`;
      this.list.appendChild(li); return;
    }
    this.filtered.forEach((opt, i) => {
      const li = document.createElement('li');
      li.className = 'ms-option' + (opt.selected ? ' selected' : '');
      const cb = document.createElement('input');
      cb.type = 'checkbox'; cb.checked = opt.selected; cb.tabIndex = -1;
      const tx = document.createElement('span');
      tx.textContent = opt.label;
      li.append(cb, tx);
      li.addEventListener('click', e => { e.stopPropagation(); this._toggleOption(opt.value); });
      li.addEventListener('mouseenter', () => { this.highlighted = i; this._updateHighlight(); });
      this.list.appendChild(li);
    });
    this._updateHighlight();
  }

  _toggleOption(value) {
    const opt = this.options.find(o => o.value === value);
    if (!opt) return;
    opt.selected = !opt.selected;
    const orig = Array.from(this.select.options).find(o => o.value === value);
    if (orig) orig.selected = opt.selected;
    this._updateUI();
    this._updateSelectAll();
    this._renderList();
    if (this.config.onChange) this.config.onChange(this.getSelectedValues());
  }

  _toggleAll() {
    const allSel = this.filtered.every(o => o.selected);
    this.filtered.forEach(o => {
      o.selected = !allSel;
      const orig = Array.from(this.select.options).find(op => op.value === o.value);
      if (orig) orig.selected = o.selected;
    });
    this._updateUI();
    this._updateSelectAll();
    this._renderList();
    if (this.config.onChange) this.config.onChange(this.getSelectedValues());
  }

  _updateSelectAll() {
    if (!this.selectAllBox) return;
    const all  = this.filtered.length > 0 && this.filtered.every(o => o.selected);
    const some = this.filtered.some(o => o.selected);
    this.selectAllBox.checked       = all;
    this.selectAllBox.indeterminate = some && !all;
  }

  _moveHighlight(dir) {
    this.highlighted = Math.max(0, Math.min(this.highlighted + dir, this.filtered.length - 1));
    this._updateHighlight();
  }

  _selectHighlighted() {
    if (this.highlighted >= 0 && this.filtered[this.highlighted])
      this._toggleOption(this.filtered[this.highlighted].value);
  }

  _updateHighlight() {
    this.list.querySelectorAll('.ms-option').forEach((el, i) => {
      el.classList.toggle('highlighted', i === this.highlighted);
      if (i === this.highlighted) el.scrollIntoView({ block:'nearest', behavior:'smooth' });
    });
  }

  _updateUI() { this._renderTags(); this._updateFooter(); this._syncHidden(); }

  _renderTags() {
    this.selectedBox.innerHTML = '';
    const sel = this.options.filter(o => o.selected);
    if (sel.length === 0) { this.selectedBox.appendChild(this.placeholder); return; }
    const visible = sel.slice(0, this.config.maxTags);
    visible.forEach(o => this.selectedBox.appendChild(this._makeTag(o)));
    if (sel.length > this.config.maxTags) {
      const cnt = document.createElement('span');
      cnt.className = 'ms-count';
      cnt.textContent = `+${sel.length - this.config.maxTags} more`;
      this.selectedBox.appendChild(cnt);
    }
  }

  _makeTag(opt) {
    const tag = document.createElement('div');
    tag.className = 'ms-tag';
    tag.innerHTML = `<span>${opt.label}</span><div class="ms-remove">×</div>`;
    tag.querySelector('.ms-remove').addEventListener('click', e => {
      e.stopPropagation(); this._toggleOption(opt.value);
    });
    return tag;
  }

  _updateFooter() {
    const n = this.options.filter(o => o.selected).length;
    this.footer.textContent = `${n} of ${this.options.length} selected`;
  }

  _syncHidden() {
    if (this.hiddenInput) this.hiddenInput.value = this.getSelectedValues().join(',');
  }

  /* ── Public API ── */
  getSelectedValues() { return this.options.filter(o => o.selected).map(o => o.value); }
  getSelectedLabels() { return this.options.filter(o => o.selected).map(o => o.label); }

  setValues(values) {
    this.options.forEach(o => {
      o.selected = values.includes(o.value);
      const orig = Array.from(this.select.options).find(op => op.value === o.value);
      if (orig) orig.selected = o.selected;
    });
    this._updateUI(); this._updateSelectAll(); this._renderList();
    if (this.config.onChange) this.config.onChange(this.getSelectedValues());
  }

  clear() { this.setValues([]); }
  destroy() { this.wrapper.remove(); }
}
Quick Setup
1 Drop the CSS and JS into your project. Add the <select multiple> with your <option> items inside a <form>.
2 Call new MultiSelect('yourSelectId', { ... }). The component hides the original select and builds its custom UI automatically.
3 Add hiddenInputId for form submission (comma-separated values). Use the onChange callback to react to selection changes.
JS API
setValues(["a","b"])
Set selection by values
getSelectedValues()
Returns array of values
getSelectedLabels()
Returns array of labels
clear() / destroy()
Reset or remove component
